69  int endIdx) {
70  // N denotes the size of the range that needs to be sorted
71  int N = endIdx - begIdx + 1;
72 
73  // Handle base cases
74  if (N <= 0 || N == 1)
75  return;
76 
77  // Pivot element is selected randomly from the range. This is some boilerplate
78  // code for generating the pivot randomly
79  std::random_device
80  rd; // Will be used to obtain a seed for the random number engine
81  std::mt19937 gen(rd()); // Standard mersenne_twister_engine seeded with rd()
82  std::uniform_int_distribution<> distrib(begIdx, endIdx);
83 
84  // Select the pivot randomly from the range list[begIdx..endIdx]
85  int pivotIdx = distrib(gen);
86  T pivot = list[pivotIdx];
87 
88  // We want to partition the list[begIdx..endIdx] as follows:
89  // 1. list[begIdx .. smallIdx] < pivot
90  // 2. list[smallIdx+1 .. bigIdx - 1] == pivot
91  // 3. list[bigIdx .. endIdx] > pivot
92 
93  // Initialize smallIdx and bigIdx
94  int smallIdx = begIdx - 1;
95  int bigIdx = endIdx + 1;
96 
97  // We will use idx to iterate over the array range
98  int idx = begIdx;
99 
100  // idx is used to iterate through list[begIdx..endIdx]
101  // At every iteration, we will check whether list[idx] is less than; greater
102  // than; or equal to the pivot element All the list elements within the range
103  // that are less than the pivot element will be moved to the beginning of the
104  // range. Ie. list[begIdx..smallIdx] will contain all the elements in range
105  // smaller than the pivot Similarly, all the list elements within the range
106  // that are greater than the pivot will be moved to the end of the range. Ie.
107  // list[bigIdx..endIdx] will contain all the elements in the range greater
108  // than the pivot list[smallIdx+1..bigIdx-1] will contain list elements ==
109  // pivot Once we have achieved the partitioning descibed above, we can run
110  // this method recursively on subranges list[begIdx..smallIdx] and
111  // list[bigIdx..endIdx]
112  while (bigIdx > smallIdx && bigIdx >= begIdx && smallIdx <= endIdx &&
113  idx < bigIdx) {
114  if (list[idx] < pivot) {
115  smallIdx++;
116  idx++;
117  } else if (list[idx] > pivot) {
118  // swap list[idx] and list[bigIdx - 1]
119  T tmp = list[idx];
120  list[idx] = list[bigIdx - 1];
121  list[bigIdx - 1] = tmp;
122 
123  // decrement bigIdx
124  bigIdx--;
125  } else {
126  // increment only idx
127  idx++;
128  }
129  }
130  // At the end of the above loop:
131  // 1. list[begIdx .. smallIdx] < pivot
132  // 2. list[smallIdx+1 .. bigIdx - 1] == pivot
133  // 3. list[bigIdx .. endIdx] > pivot
134 
135  // Run quicksort on list[begIdx .. smallIdx] and list[bigIdx .. endIdx]
136  this->sortUtilNumeric(list, begIdx, smallIdx);
137  this->sortUtilNumeric(list, bigIdx, endIdx);
138 }
