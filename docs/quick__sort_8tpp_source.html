11  // N denotes the size of the range that needs to be sorted
12  int N = endIdx - begIdx + 1;
13 
14  // Handle base cases
15  if (N <= 0 || N == 1)
16  return;
17 
18  // Pivot element is selected randomly from the range. This is some boilerplate
19  // code for generating the pivot randomly
20  std::random_device
21  rd; // Will be used to obtain a seed for the random number engine
22  std::mt19937 gen(rd()); // Standard mersenne_twister_engine seeded with rd()
23  std::uniform_int_distribution<> distrib(begIdx, endIdx);
24 
25  // Select the pivot randomly from the range list[begIdx..endIdx]
26  int pivotIdx = distrib(gen);
27  T pivot = list[pivotIdx];
28  LOG_DEBUG << "Pivot elem: " << pivot;
29 
30  // We want to partition the list[begIdx..endIdx] as follows:
31  // 1. list[begIdx .. smallIdx] < pivot
32  // 2. list[smallIdx+1 .. bigIdx - 1] == pivot
33  // 3. list[bigIdx .. endIdx] > pivot
34 
35  // Initialize smallIdx and bigIdx
36  int smallIdx = begIdx - 1;
37  int bigIdx = endIdx + 1;
38 
39  // We will use idx to iterate over the array range
40  int idx = begIdx;
41 
42  // idx is used to iterate through list[begIdx..endIdx]
43  // At every iteration, we will check whether list[idx] is less than; greater
44  // than; or equal to the pivot element All the list elements within the range
45  // that are less than the pivot element will be moved to the beginning of the
46  // range. Ie. list[begIdx..smallIdx] will contain all the elements in range
47  // smaller than the pivot Similarly, all the list elements within the range
48  // that are greater than the pivot will be moved to the end of the range. Ie.
49  // list[bigIdx..endIdx] will contain all the elements in the range greater
50  // than the pivot list[smallIdx+1..bigIdx-1] will contain list elements ==
51  // pivot Once we have achieved the partitioning descibed above, we can run
52  // this method recursively on subranges list[begIdx..smallIdx] and
53  // list[bigIdx..endIdx]
54  while (bigIdx > smallIdx && bigIdx >= begIdx && smallIdx <= endIdx &&
55  idx < bigIdx) {
56  if (list[idx] < pivot) {
57  // swap list[idx] and list[smallIdx + 1]
58  T tmp = list[idx];
59  list[idx] = list[smallIdx + 1];
60  list[smallIdx + 1] = tmp;
61 
62  // increment smallIdx
63  smallIdx++;
64 
65  // increment idx
66  idx++;
67  } else if (list[idx] > pivot) {
68  // swap list[idx] and list[bigIdx - 1]
69  T tmp = list[idx];
70  list[idx] = list[bigIdx - 1];
71  list[bigIdx - 1] = tmp;
72 
73  // decrement bigIdx
74  bigIdx--;
75  } else {
76  // increment only idx
77  idx++;
78  }
79  }
80  // At the end of the above loop:
81  // 1. list[begIdx .. smallIdx] < pivot
82  // 2. list[smallIdx+1 .. bigIdx - 1] == pivot
83  // 3. list[bigIdx .. endIdx] > pivot
84  std::vector<T> range(list.begin() + begIdx, list.begin() + endIdx + 1);
85  LOG_DEBUG << "Range after partitioning: " << logging::str(range);
86 
87  // Run quicksort on list[begIdx .. smallIdx] and list[bigIdx .. endIdx]
88  this->sortUtilNumeric(list, begIdx, smallIdx);
89  this->sortUtilNumeric(list, bigIdx, endIdx);
90 }
